
in vec3 surfaceNormal;
in vec3 toLightVector;
in vec2 texCoords;

uniform sampler2D tex;

float ambientLighting = 0.1;
float tiling = 25.0;
void main() {
	float nDotl = dot(normalize(surfaceNormal), normalize(toLightVector));
	float brightness = max(nDotl, ambientLighting);
	gl_FragColor = vec4(brightness,brightness,brightness, 1.0) * texture2D(tex, texCoords * tiling);
};
